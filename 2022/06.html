library(slider)

start_distinct_marker <- function(data, n) {
    data %>% 
        str_split_1("") %>%
        slide(~.x, .before = n - 1) %>% 
        map(function(x){
            n_distinct(x) == n
        }) %>% 
        which.max()
}
